    .text
    .globl main

main:
    addi $t4, $zero, 10   

    ble  $t4, $zero, return_n

    addi $t0, $zero, 0
    addi $t1, $zero, 1  
    addi $t2, $zero, 0  

loop:
    beq  $t2, $t4, done 
    add  $t3, $zero, $t1
    add  $t1, $t0, $t1  
    add  $t0, $zero, $t3 
    addi $t2, $t2, 1     

    j loop

done:
    add  $v0, $zero, $t1 

halt:  
    beq  $zero, $zero, halt 

return_n:
    add  $v0, $zero, $t4  
    j halt
